    <h2>Tabuada</h2>

    <button onclick="mostrarTabuada()">Mostrar tabuada</button>
    <button onclick="limpar()">Limpar tabuada</button>

    <div id="output"></div>

    <script>
        const output = document.getElementById("output");
        
        function mostrarTabuada(){
            //pedir para o usuario digitar um número
            let n = parseFloat(prompt("digite um número"));
            /*dessa forma estara convertendo o valor para 
            número*/

            let msg = "";
            for(i=1; i<=1000; i++){
                //1x1 = 
                msg += `${n} x ${i} = ${(n * i)} <br>`;
            }
            output.innerHTML = msg;
        }

        function limpar(){
            output.innerHTML = "";
            /*A propriedade innerHTML é usada para obter ou definir o 
            conteúdo HTML dentro de um elemento. Isso significa que 
            ele inclui tags HTML e, portanto, permite que você insira 
            HTML formatado dentro do elemento.
            
            
            A propriedade innerText é usada para obter ou definir o texto visível 
            dentro de um elemento, sem interpretar o conteúdo como HTML. 
            Isso significa que qualquer marcação HTML presente no texto será 
            tratada como texto simples e não será renderizada como elementos HTML.
            Ao definir o conteúdo com innerText, você está apenas substituindo 
            o texto visível e mantendo qualquer estrutura HTML subjacente intacta.*/
        }
    </script>
